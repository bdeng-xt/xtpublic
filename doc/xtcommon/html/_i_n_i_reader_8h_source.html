127  /* Strip whitespace chars off end of given string, in place. Return s. */
128  inline static char* rstrip(char* s)
129  {
130  char* p = s + strlen(s);
131  while (p > s && isspace((unsigned char)(*--p)))
132  *p = '\0';
133  return s;
134  }

136  /* Return pointer to first non-whitespace char in given string. */
137  inline static char* lskip(const char* s)
138  {
139  while (*s && isspace((unsigned char)(*s)))
140  s++;

144  /* Return pointer to first char (of chars) or inline comment in given string,
145  or pointer to null at end of string if neither found. Inline comment must
146  be prefixed by a whitespace character to register as a comment. */
147  inline static char* find_chars_or_comment(const char* s, const char* chars)
148  {
149 #if INI_ALLOW_INLINE_COMMENTS
150  int was_space = 0;
151  while (*s && (!chars || !strchr(chars, *s)) &&
152  !(was_space && strchr(INI_INLINE_COMMENT_PREFIXES, *s))) {
153  was_space = isspace((unsigned char)(*s));
154  s++;
155  }
156 #else
157  while (*s && (!chars || !strchr(chars, *s))) {
158  s++;
159  }
160 #endif
161  return (char*)s;
162  }

173  inline int ini_parse_stream(ini_reader reader, void* stream, ini_handler handler,
174  void* user)
175  {
176  /* Uses a fair bit of stack (use heap instead if you need to) */
177 #if INI_USE_STACK
178  char line[INI_MAX_LINE];
179 #else
180  char* line;
181 #endif
182  char section[MAX_SECTION] = "";
183  char prev_name[MAX_NAME] = "";
184 
185  char* start;
186  char* end;
187  char* name;
188  char* value;
189  int lineno = 0;
190  int error = 0;
191 
192 #if !INI_USE_STACK
193  line = (char*)malloc(INI_MAX_LINE);
194  if (!line) {
195  return -2;
196  }
197 #endif
198 
199  /* Scan through stream line by line */
200  while (reader(line, INI_MAX_LINE, stream) != NULL) {
201  lineno++;
202 
203  start = line;
204 #if INI_ALLOW_BOM
205  if (lineno == 1 && (unsigned char)start[0] == 0xEF &&
206  (unsigned char)start[1] == 0xBB &&
207  (unsigned char)start[2] == 0xBF) {
208  start += 3;
209  }
210 #endif
211  start = lskip(rstrip(start));
212 
213  if (*start == ';' || *start == '#') {
214  /* Per Python configparser, allow both ; and # comments at the
215  start of a line */
216  }
217 #if INI_ALLOW_MULTILINE
218  else if (*prev_name && *start && start > line) {
219 
220 #if INI_ALLOW_INLINE_COMMENTS
221  end = find_chars_or_comment(start, NULL);
222  if (*end)
223  *end = '\0';
224  rstrip(start);
225 #endif
226 
227  /* Non-blank line with leading whitespace, treat as continuation
228  of previous name's value (as per Python configparser). */
229  if (!handler(user, section, prev_name, start) && !error)
230  error = lineno;
231  }
232 #endif
233  else if (*start == '[') {
234  /* A "[section]" line */
235  end = find_chars_or_comment(start + 1, "]");
236  if (*end == ']') {
237  *end = '\0';
238  strncpy0(section, start + 1, sizeof(section));
239  *prev_name = '\0';
240  }
241  else if (!error) {
242  /* No ']' found on section line */
243  error = lineno;
244  }
245  }
246  else if (*start) {
247  /* Not a comment, must be a name[=:]value pair */
248  end = find_chars_or_comment(start, "=:");
249  if (*end == '=' || *end == ':') {
250  *end = '\0';
251  name = rstrip(start);
252  value = lskip(end + 1);
253 #if INI_ALLOW_INLINE_COMMENTS
254  end = find_chars_or_comment(value, NULL);
255  if (*end)
256  *end = '\0';
257 #endif
258  rstrip(value);
259 
260  /* Valid name[=:]value pair found, call handler */
261  strncpy0(prev_name, name, sizeof(prev_name));
262  if (!handler(user, section, name, value) && !error)
263  error = lineno;
264  }
265  else if (!error) {
266  /* No '=' or ':' found on name[=:]value line */
267  error = lineno;
268  }
269  }
270 
271 #if INI_STOP_ON_FIRST_ERROR
272  if (error)
273  break;
274 #endif
275  }
276 
277 #if !INI_USE_STACK
278  free(line);
279 #endif
280 
281  return error;
282  }
