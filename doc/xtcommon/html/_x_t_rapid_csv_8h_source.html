157  void ToVal(const std::string& pStr, T& pVal) const
158  {
159  try
160  {
161  if (typeid(T) == typeid(int))
162  {
163  pVal = static_cast<T>(std::stoi(pStr));
164  return;
165  }
166  else if (typeid(T) == typeid(long))
167  {
168  pVal = static_cast<T>(std::stol(pStr));
169  return;
170  }
171  else if (typeid(T) == typeid(long long))
172  {
173  pVal = static_cast<T>(std::stoll(pStr));
174  return;
175  }
176  else if (typeid(T) == typeid(unsigned))
177  {
178  pVal = static_cast<T>(std::stoul(pStr));
179  return;
180  }
181  else if (typeid(T) == typeid(unsigned long))
182  {
183  pVal = static_cast<T>(std::stoul(pStr));
184  return;
185  }
186  else if (typeid(T) == typeid(unsigned long long))
187  {
188  pVal = static_cast<T>(std::stoull(pStr));
189  return;
190  }
191  }
192  catch (...)
193  {
194  if (!mCsvConverterParams.mHasDefaultConverter)
195  {
196  throw;
197  }
198  else
199  {
200  pVal = static_cast<T>(mCsvConverterParams.mDefaultInteger);
201  return;
202  }
203  }
204 
205  try
206  {
207  if (typeid(T) == typeid(float))
208  {
209  pVal = static_cast<T>(std::stof(pStr));
210  return;
211  }
212  else if (typeid(T) == typeid(double))
213  {
214  pVal = static_cast<T>(std::stod(pStr));
215  return;
216  }
217  else if (typeid(T) == typeid(long double))
218  {
219  pVal = static_cast<T>(std::stold(pStr));
220  return;
221  }
222  }
223  catch (...)
224  {
225  if (!mCsvConverterParams.mHasDefaultConverter)
226  {
227  throw;
228  }
229  else
230  {
231  pVal = static_cast<T>(mCsvConverterParams.mDefaultFloat);
232  return;
233  }
234  }
235 
236  if (typeid(T) == typeid(char))
237  {
238  pVal = static_cast<T>(pStr[0]);
239  return;
240  }
241  else
242  {
243  throw no_csv_converter();
244  }

1105  void ReadCsv(std::istream& pStream)
1106  {
1107  pStream.seekg(0, std::ios::end);
1108  std::streamsize fileLength = pStream.tellg();
1109  pStream.seekg(0, std::ios::beg);
1110  const std::streamsize bufLength = 64 * 1024;
1111  std::vector<char> buffer(bufLength);
1112  std::vector<std::string> row;
1113  std::string cell;
1114  bool quoted = false;
1115  int cr = 0;
1116  int lf = 0;
1117 
1118  while (fileLength > 0)
1119  {
1120  std::streamsize readLength = std::min(fileLength, bufLength);
1121  pStream.read(buffer.data(), readLength);
1122  for (int i = 0; i < readLength; ++i)
1123  {
1124  if (buffer[i] == '"')
1125  {
1126  if (cell.empty() || cell[0] == '"')
1127  {
1128  quoted = !quoted;
1129  }
1130  cell += buffer[i];
1131  }
1132  else if (buffer[i] == mCsvSeparatorParams.mSeparator)
1133  {
1134  if (!quoted)
1135  {
1136  row.push_back(mCsvSeparatorParams.mTrim ? Trim(cell) : cell);
1137  cell.clear();
1138  }
1139  else
1140  {
1141  cell += buffer[i];
1142  }
1143  }
1144  else if (buffer[i] == '\r')
1145  {
1146  ++cr;
1147  }
1148  else if (buffer[i] == '\n')
1149  {
1150  ++lf;
1151  row.push_back(mCsvSeparatorParams.mTrim ? Trim(cell) : cell);
1152  cell.clear();
1153  mData.push_back(row);
1154  row.clear();
1155  quoted = false; // disallow line breaks in quoted string, by auto-unquote at linebreak
1156  }
1157  else
1158  {
1159  cell += buffer[i];
1160  }
1161  }
1162  fileLength -= readLength;
1163  }
1164 
1165  // Handle last line without linebreak
1166  if (!cell.empty() || !row.empty())
1167  {
1168  row.push_back(mCsvSeparatorParams.mTrim ? Trim(cell) : cell);
1169  cell.clear();
1170  mData.push_back(row);
1171  row.clear();
1172  }
1173 
1174  // Assume CR/LF if at least half the linebreaks have CR
1175  mCsvSeparatorParams.mHasCR = (cr > (lf / 2));
1176 
1177  // Set up column labels
1178  if ((mCsvLabelParams.mColumnNameIdx >= 0) &&
1179  (mData.size() > 0))
1180  {
1181  int i = 0;
1182  for (auto& columnName : mData[mCsvLabelParams.mColumnNameIdx])
1183  {
1184  mColumnNames[columnName] = i++;
1185  }
1186  }
1187 
1188  // Set up row labels
1189  if ((mCsvLabelParams.mRowNameIdx >= 0) &&
1190  (static_cast<int>(mData.size()) >
1191  (mCsvLabelParams.mColumnNameIdx + 1)))
1192  {
1193  int i = 0;
1194  for (auto& dataRow : mData)
1195  {
1196  mRowNames[dataRow[mCsvLabelParams.mRowNameIdx]] = i++;
1197  }
1198  }
1199  }
