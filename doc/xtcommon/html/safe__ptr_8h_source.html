250  bool success;

252  template<typename mtx_t>
253  std::unique_lock<mtx_t> try_lock_one(mtx_t &mtx) const {
254  std::unique_lock<mtx_t> lock(mtx, std::defer_lock_t());
255  for (size_t i = 0; i < spin_iterations; ++i) if (lock.try_lock()) return lock;
256  const std::chrono::steady_clock::time_point start_time = std::chrono::steady_clock::now();
257  //while (!lock.try_lock_for(duration(deadlock_timeout))) // only for timed mutexes
258  while (!lock.try_lock()) {
259  auto const time_remained = duration(deadlock_timeout) - std::chrono::duration_cast<duration>(std::chrono::steady_clock::now() - start_time);
260  if (time_remained <= duration(0))
261  break;
262  else
263  std::this_thread::sleep_for(time_remained);
264  }
265  return lock;
266  }

269  std::shared_ptr<std::unique_lock<mtx_t>> try_lock_ptr_one(mtx_t &mtx) const {
270  return std::make_shared<std::unique_lock<mtx_t>>(try_lock_one(mtx));
271  }
272 
273  public:
274  template<typename... Args>
275  lock_timed_any(Args& ...args) {
276  do {
277  success = true;
278  for (auto &lock_ptr : { try_lock_ptr_one(*args.mtx_ptr.get()) ... }) {
279  locks_ptr_vec.emplace_back(lock_ptr);
280  if (!lock_ptr->owns_lock()) {
281  success = false;
282  locks_ptr_vec.clear();
283  std::this_thread::sleep_for(duration(deadlock_timeout));
284  break;
285  }
286  }
287  } while (!success && lock_count == lock_count_t::lock_infinity);
288  }
